```yaml
    branches: [ main ]

env:
  CARGO_TERM_COLOR: always

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        rust: [stable, nightly]
        features: [["default"], ["mode_phi"], ["mode_one"], ["mode_zero"]]

    steps:
    - uses: actions/checkout@v3
    
    - name: Install Rust
      uses: actions-rs/toolchain@v1
      with:
        toolchain: ${{ matrix.rust }}
        override: true
        components: rustfmt, clippy, miri
    
    - name: Cache cargo registry
      uses: actions/cache@v3
      with:
        path: |
          ~/.cargo/registry
          ~/.cargo/git
        key: ${{ runner.os }}-cargo-${{ hashFiles('**/Cargo.lock') }}
    
    - name: Build
      run: cargo build --verbose --features "${{ matrix.features }}"
    
    - name: Run tests
      run: cargo test --verbose --features "${{ matrix.features }}"
    
    - name: Run clippy
      run: cargo clippy --verbose --features "${{ matrix.features }}" -- -D warnings
    
    - name: Run rustfmt
      run: cargo fmt --verbose -- --check
    
    - name: Security audit
      run: cargo audit
    
    - name: Run MIRI (nightly only)
      if: matrix.rust == 'nightly'
      run: cargo miri test --features "${{ matrix.features }}"

  verify:
    runs-on: ubuntu-latest
    needs: test
    steps:
    - uses: actions/checkout@v3
    
    - name: Run formal verification tools
      run: |
        cd src/kernel
        cargo kani --harness critical_functions
        cargo verify
    
    - name: Generate verification report
      run: ./scripts/generate-verification-report.sh

  performance:
    runs-on: ubuntu-latest
    needs: test
    steps:
    - uses: actions/checkout@v3
    
    - name: Run benchmarks
      run: |
        cd src/kernel
        cargo bench --features "mode_phi"
        cargo bench --features "mode_one"
        cargo bench --features "mode_zero"
    
    - name: Upload benchmark results
      uses: actions/upload-artifact@v3
      with:
        name: benchmark-results
        path: target/criterion

  security:
    runs-on: ubuntu-latest
    needs: test
    steps:
    - uses: actions/checkout@v3
    
    - name: Run security tests
      run: |
        cd tests/security
        cargo test --release --features "security_tests"
        ./run-side-channel-tests.sh
    
    - name: Run quantum attack simulations
      run: |
        cd src/tools/qats
        cargo run --release -- --attack shor
        cargo run --release -- --attack grover
        cargo run --release -- --attack annealing
```
